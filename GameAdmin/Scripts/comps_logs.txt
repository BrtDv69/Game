/* Table: COMPS_LOGS */

CREATE TABLE COMPS_LOGS (
    CHANGE_TIME TIMESTAMP NOT NULL,
    OLD_VALID SMALLINT NOT NULL,
    NEW_VALID SMALLINT NOT NULL,
    COMP_ID INTEGER NOT NULL);





/* Foreign keys definition */

ALTER TABLE COMPS_LOGS ADD CONSTRAINT FK_COMPS_LOGS FOREIGN KEY (COMP_ID) REFERENCES COMPS (ID) ON UPDATE CASCADE;


/* Indices definition */

CREATE INDEX COMPS_LOGS_IDX1 ON COMPS_LOGS (OLD_VALID, NEW_VALID);
CREATE INDEX COMPS_LOGS_IDX2 ON COMPS_LOGS (CHANGE_TIME);
