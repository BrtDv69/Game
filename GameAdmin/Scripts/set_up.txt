/* Table: SET_UP */

CREATE TABLE SET_UP (
    IP_ADDRESS VARCHAR(15),
    HASH_IMAGE INTEGER NOT NULL,
    PORT INTEGER,
    POST_PORT INTEGER DEFAULT 25,
    POST_USER_NAME VARCHAR(20),
    POST_ADDRESS VARCHAR(100),
    DESTINATION_ADDRESS VARCHAR(100),
    CURRENCY_DECIMALS INTEGER DEFAULT 100 NOT NULL,
    CHANGE_TIME TIMESTAMP DEFAULT 'now' NOT NULL,
    IMAGE BLOB SUB_TYPE 2 SEGMENT SIZE 1,
    ON_ORDERLESS SMALLINT NOT NULL);



/* Triggers definition */


SET TERM ^ ;





/* Trigger: SET_UP_BU0 */
CREATE TRIGGER SET_UP_BU0 FOR SET_UP
ACTIVE BEFORE UPDATE POSITION 0
as
begin
 NEW.CHANGE_TIME = 'now';
end
^


SET TERM ; ^
