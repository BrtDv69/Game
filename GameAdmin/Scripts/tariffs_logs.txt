/* Table: TARIFFS_LOGS */

CREATE TABLE TARIFFS_LOGS (
    CHANGE_TIME TIMESTAMP NOT NULL,
    OLD_PRICE DOUBLE PRECISION NOT NULL,
    NEW_PRICE DOUBLE PRECISION NOT NULL,
    TARIFF_ID INTEGER NOT NULL);





/* Foreign keys definition */

ALTER TABLE TARIFFS_LOGS ADD CONSTRAINT FK_TARIFFS_LOGS FOREIGN KEY (TARIFF_ID) REFERENCES TARIFFS (ID) ON UPDATE CASCADE;


/* Indices definition */

CREATE INDEX TARIFFS_LOGS_IDX1 ON TARIFFS_LOGS (OLD_PRICE, NEW_PRICE);
CREATE INDEX TARIFFS_LOGS_IDX2 ON TARIFFS_LOGS (CHANGE_TIME);
