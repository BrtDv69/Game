/* Table: UNORDER_TIME */

CREATE TABLE UNORDER_TIME (
    REC_ID INTEGER NOT NULL,
    COMP_ID INTEGER NOT NULL,
    TIME_START TIMESTAMP NOT NULL,
    TIME_STOP TIMESTAMP NOT NULL);





/* Foreign keys definition */

ALTER TABLE UNORDER_TIME ADD CONSTRAINT FK_UNORDER_TIME FOREIGN KEY (COMP_ID) REFERENCES COMPS (ID) ON UPDATE CASCADE;


/* Indices definition */

CREATE INDEX UNORDER_TIME_IDX1 ON UNORDER_TIME (TIME_STOP, TIME_START);
CREATE UNIQUE INDEX UNORDER_TIME_IDX2 ON UNORDER_TIME (REC_ID);
